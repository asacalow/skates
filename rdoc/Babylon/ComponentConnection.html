Babylon::ComponentConnection

ComponentConnection is in charge of the XMPP connection itself. Upon stanza reception, and depending on the status (connected... etc), this component will handle or forward the stanzas.

33:     def receive_stanza(stanza)
34:       case @state
35:       when :connected # Most frequent case
36:           super # Can be dispatched
37:           
38:       when :wait_for_stream
39:         if stanza.name == "stream:stream" && stanza.attributes['id']
40:           # This means the XMPP session started!
41:           # We must send the handshake now.
42:           send(handshake(stanza))
43:           @state = :wait_for_handshake
44:         else
45:           raise
46:         end
47: 
48:       when :wait_for_handshake
49:         if stanza.name == "handshake"
50:           begin
51:             @handler.on_connected(self) if @handler and @handler.respond_to?("on_connected")
52:           rescue
53:             Babylon.logger.error("on_connected failed : #{$!}\n#{$!.backtrace.join("\n")}")
54:           end
55:           @state = :connected
