47:     def route(stanza)
48:       return false if !@@connection
49:       @routes ||= []
50:       @routes.each { |route|
51:         if route.accepts?(stanza)
52:           # Here should happen the magic : call the controller
53:           Babylon.logger.info("ROUTING TO #{route.controller}::#{route.action}")
54:           # Parsing the stanza
55:           begin
56:             controller = route.controller.new({:stanza => Kernel.const_get(route.action.capitalize).new(stanza)})
57:           rescue NameError

59:           end
